[key0]
; group is the key group number. Keys in the same group behave like a
; radio button group, when one is pressed the others turn off
; group 0 means no group. Groups are from 1 to 31.
group = [1]

; longgroup is the same as group but for long press cycles.
; Keys in the same longgroup behave like a radio button group for long presses.
longgroup = [1]

; cycle is the number of steps in the key cycle from 0 to 9. 
; If 1, the key does not cycle.
cycle = [3]

; long press can control an independent cycle. If 0 or undefined, it
; defaults at no cycle, >0 is the number of steps.
longcycle = [2]

; stomp = 0 - no effect on the 6 sub-panels in main visualization
; stomp = 1 - reflect cycle
; stomp = 2 - reflect longcycle
; with this, you can have a feedback of the button state on the corresponding
; 3x2 blocks of the 6 available in main visualization. You can have the button
; show either the main or the sub (longpress) cycle. The background color will
; be the first non-null color of the LED definition. While the 3 LEDS provide
; reference, using blocks allows you to use labels (although limited to 5 chars). 
stompmode = [2]

; ledX is the color of the key LED per cycle step. X is the cycle step number.
; a null color ([] or [-]) means the LED is unchanged.
; black color (0x000000) means the LED is off.
; ledXl is the same for the long press cycle step. 
led1 = [0x00ff00][-][0x00ff00]
led2 = [0x0000ff][-][0x0000ff]
led3 = [0x0000ff][-][0x00ff00]
led1l = [-][0xffffff][-]
led2l = [-][0xff00ff][-]

; labelX is the text on the key per cycle step. X is the cycle step number.
; use label1 when there's no cycle. A colon (:) in the label creates a line 
; break, but only for a max of 2 rows and 2x8 characters.

label2 = [Foo]
label3 = [Bar]
; labelXl is the same for the long press cycle step. This label is shown

label2l = [B OFF]

; ===================================================================
; VIRTUAL KEYS (6-31)
; ===================================================================
; Keys 6-31 are virtual: no physical button, no LEDs.
; They are triggered via ext_capture_cc (see CAPTURE.md) or
; [KEY][n] commands from other keys or init_commands.
;
; Virtual keys support all properties EXCEPT led (ledX, ledXl)
; since there are no physical LEDs. They DO support:
; group, longgroup, cycle, longcycle, stompmode, labels, and
; all key commands (keyXdn, keyXup, keyXldn, keyXlup).
;
; Note: stompmode on virtual keys may not map to a display
; sub-panel (the 6 panels correspond to physical keys 0-5).
;
; Virtual keys don't need to be contiguous � define only the
; ones you need.
;
; To trigger key 6 via ext_capture_cc (e.g. ch1/CC#30):
;   press:          send CC#30 value 6   (0x06)
;   long press:     send CC#30 value 38  (0x26)
;   key up:         send CC#30 value 70  (0x46)
;   long press up:  send CC#30 value 102 (0x66)

[key6]
; example: external footswitch toggles delay on/off
group = [1]
cycle = [2]
label1 = [DLY ON]
label2 = [DLY OF]
key1dn = [1][CC][2][127]
key2dn = [1][CC][2][0]
